\section*{[2024 EXAM] Problem 3: Upwind disc. Galerkin method for lin. ODE systems in t}

We are given $u \in [C^1([0,T])]^{D}$ solving the $D$--dimensional linear ODE system
\[
    u_t(t) + A u(t) = 0 \quad\text{for } t\in(0,T), \tag{5}
\]
with initial data
\[
    u(0) = u_0 \in \mathbb{R}^D,
\]
where $A\in\mathbb{R}^{D\times D}$ is a constant symmetric matrix.
Let $(\cdot,\cdot)$ and $|\cdot|$ denote the Euclidean inner product and norm on $\mathbb{R}^D$.
\subsection*{(a) Variational (strong) formulation and spaces}
The \emph{strong} formulation is simply:
\[
    \begin{cases}
        u_t(t) + A u(t) = 0 & \text{for all } t\in(0,T), \\[2mm]
        u(0) = u_0,         &
    \end{cases}
\]
with $u \in C^1([0,T];\mathbb{R}^D)$.
A natural \emph{variational} formulation is obtained by testing with an arbitrary constant
vector $v\in\mathbb{R}^D$ and integrating in time over $(0,T)$:
\[
    \int_0^T (u_t(t) + A u(t), v)\,dt = 0
    \quad\forall v\in\mathbb{R}^D.
\]
Equivalently,
\[
    \int_0^T (u_t(t), v)\,dt + \int_0^T (A u(t), v)\,dt = 0
    \quad\forall v\in\mathbb{R}^D.
\]
\textbf{Choice of spaces.}
\begin{itemize}
    \item The solution is sought in $u \in C^1([0,T];\mathbb{R}^D)$ so that $u_t$ exists
          and is continuous, making $\int_0^T (u_t,v)\,dt$ well defined.
    \item Since $A$ is a bounded linear operator on $\mathbb{R}^D$,
          $t\mapsto Au(t)$ is continuous and $\int_0^T (Au,v)\,dt$ is also well defined.
    \item The test space is $V = \mathbb{R}^D$ (time--independent vectors), which is natural
          since the problem is purely in time with spatial dimension $D$.
\end{itemize}
Thus, the variational (strong) formulation is:
Find  $u \in C^1([0,T];\mathbb{R}^D)$ with $u(0)=u_0$ s.t.
\[
    \boxed{
        \int_0^T \bigl( u_t(t), v \bigr)\,dt
        +
        \int_0^T \bigl( A u(t), v\bigr)\,dt = 0
        \quad\forall v\in\mathbb{R}^D.
    }
\]
This formulation is equivalent to the pointwise ODE (5) by the fundamental theorem
of calculus (differentiate the identity with respect to $T$ and use arbitrariness of $v$).
\subsection*{Notation for the dG time space}
We partition $[0,T]$ by
\[
    0 = t_0 < t_1 < \dots < t_N = T,\qquad I_n := (t_n,t_{n+1}),
\]
and define the discontinuous Galerkin space in time
\[
    V_k^{\mathrm{dG}}([0,T])
    :=
    \left\{
    v : [0,T]\to\mathbb{R}^D \,\middle|\,
    v|_{I_n} \in \mathbb{P}_k(I_n;\mathbb{R}^D),\ 0\le n\le N-1
    \right\},
\]
i.e.\ piecewise polynomials of degree $\le k$ in time, allowed to jump at the nodes $t_n$.
For $V\in V_k^{\mathrm{dG}}$, we denote left and right limits at $t_n$ by
\[
    V_n^- := \lim_{t\to t_n^-} V(t),\qquad
    V_n^+ := \lim_{t\to t_n^+} V(t),
\]
and define the jump
\[
    \jump{V_n} := V_n^+ - V_n^-.
\]
We are given the \emph{upwind dG} formulation: find
$U\in V_k^{\mathrm{dG}}([0,T])$ such that
\[
    \sum_{n=0}^{N-1} \int_{I_n} U_t \cdot V\,dt
    + \jump{U_n}\cdot V_n^+
    + \int_{I_n} A U\cdot V\,dt
    = 0
    \quad\forall V\in V_k^{\mathrm{dG}}([0,T]), \tag{6}
\]
with $U_0^- := u_0$ and $\jump{U_n} := U_n^+ - U_n^-$.
\subsection*{(b) Energy inequality for arbitrary $W\in V_k^{\mathrm{dG}}$}
We must show that for any $W\in V_k^{\mathrm{dG}}([0,T])$,
\[
    \int_{I_n} W_t\cdot W\,dt
    + \jump{W_n}\cdot W_n^+
    \;\ge\;
    \frac12 |W_{n+1}^-|^2 - \frac12 |W_n^-|^2.
    \tag{7}
\]
Fix $n$ and work on a single interval $I_n=(t_n,t_{n+1})$.

\textbf{Step 1: Term $\displaystyle\int_{I_n} W_t\cdot W$.}
On $I_n$, $W$ is smooth (a polynomial), so
\[
    \frac{d}{dt}\,\frac12 |W(t)|^2 = W_t(t)\cdot W(t).
\]
Integrate over $I_n$:
\[
    \int_{I_n} W_t\cdot W\,dt
    = \int_{t_n}^{t_{n+1}} \frac{d}{dt}\Bigl(\frac12 |W(t)|^2\Bigr)\,dt
    = \frac12 |W_{n+1}^-|^2 - \frac12 |W_n^+|^2,
\]
where $W_n^+ := \lim_{t\to t_n^+} W(t)$ and $W_{n+1}^- := \lim_{t\to t_{n+1}^-} W(t)$.

\textbf{Step 2: Jump term.}
By definition,
\[
    \jump{W_n}\cdot W_n^+
    = (W_n^+ - W_n^-)\cdot W_n^+
    = |W_n^+|^2 - W_n^-\cdot W_n^+.
\]
\textbf{Step 3: Combine and estimate.}
Add the two contributions:
\begin{align*}
    \int_{I_n} W_t\cdot W\,dt + \jump{W_n}\cdot W_n^+
     & = \Bigl(\frac12 |W_{n+1}^-|^2 - \frac12 |W_n^+|^2\Bigr)
    + \bigl(|W_n^+|^2 - W_n^-\cdot W_n^+\bigr)                 \\
     & = \frac12 |W_{n+1}^-|^2
    + \frac12 |W_n^+|^2 - W_n^-\cdot W_n^+.
\end{align*}
We compare this with $\frac12 |W_{n+1}^-|^2 - \frac12 |W_n^-|^2$.
Subtract the latter from both sides:
\begin{align*}
     & \left(
    \int_{I_n} W_t\cdot W\,dt + \jump{W_n}\cdot W_n^+
    \right)
    - \left(\frac12 |W_{n+1}^-|^2 - \frac12 |W_n^-|^2\right)           \\
     & = \frac12 |W_n^+|^2 - W_n^-\cdot W_n^+ + \frac12 |W_n^-|^2      \\
     & = \frac12\Bigl(|W_n^+|^2 - 2 W_n^-\cdot W_n^+ + |W_n^-|^2\Bigr) \\
     & = \frac12\,|W_n^+ - W_n^-|^2 \;\ge\; 0.
\end{align*}
Hence
\[
    \int_{I_n} W_t\cdot W\,dt + \jump{W_n}\cdot W_n^+
    \;\ge\;
    \frac12 |W_{n+1}^-|^2 - \frac12 |W_n^-|^2,
\]
which is exactly (7).
\subsection*{Interpolation operator and its estimate}
We are given the interpolation operator
\[
    I : \bigl(C^0([0,T])\cap L^2([0,T])\bigr)^D \to V_k^{\mathrm{dG}}([0,T])
\]
defined by
\[
    \int_{I_n} (I(w) - w)\cdot V\,dt = 0
    \quad\forall V\in \mathbb{P}_{k-1}(I_n;\mathbb{R}^D),
\]
and
\[
    I(w)(t_n) = w(t_n),\qquad n=0,\dots,N-1,
\]
for $k\ge1$. If $w\in H^{k+1}(I_n)$, then
\[
    \|w - I(w)\|_{L^2(I_n)} \le C \tau^{k+1} |w|_{H^{k+1}(I_n)}, \tag{8}
\]
with $\tau := \max_n (t_{n+1}-t_n)$.
\subsection*{(c) A priori error estimate for the upwind dG method}
Assume that $A$ induces a continuous and coercive norm, i.e.\ there exist constants
$0<\alpha\le\beta$ such that
\[
    \alpha|z|^2 \le (A z,z) \le \beta |z|^2 \quad\forall z\in\mathbb{R}^D.
\]
Let $u$ solve the continuous problem (5), and $U$ be the upwind dG solution (6).
We want to show
\[
    |U_N^- - u_N^-|^2 \;\le\; C\,\tau^{2k+2}\,|u|_{H^{k+1}(0,T)}^2. \tag{9}
\]
\textbf{Error splitting.}
Set the (total) error
\[
    e := U - u.
\]
Split it into
\[
    e = (U - I(u)) + (I(u) - u) =: \xi + \eta,
\]
where $\xi := U - I(u) \in V_k^{\mathrm{dG}}$ is the discrete error and
$\eta := I(u) - u$ is the interpolation error.
We are given the \emph{error equation}
\[
    \int_{I_n} (U_t - u_t)\cdot V\,dt
    + \jump{U_n - u_n}\cdot V_n^+
    + \int_{I_n} A (U - u)\cdot V\,dt
    = 0
    \quad\forall V\in V_k^{\mathrm{dG}}. \tag{10}
\]
Substitute $U - u = \xi + \eta$ and $U_t - u_t = \xi_t + \eta_t$:
\[
    \int_{I_n} (\xi_t + \eta_t)\cdot V\,dt
    + \jump{\xi_n + \eta_n}\cdot V_n^+
    + \int_{I_n} A (\xi + \eta)\cdot V\,dt
    = 0 \quad\forall V\in V_k^{\mathrm{dG}}.
\]
\textbf{Choose the test function $V = \xi$.}
For $V=\xi$ we obtain, on each $I_n$,
\[
    \int_{I_n} \xi_t\cdot \xi\,dt
    + \jump{\xi_n}\cdot \xi_n^+
    + \int_{I_n} A\xi\cdot\xi\,dt
    = -\int_{I_n} \eta_t\cdot \xi\,dt
    - \jump{\eta_n}\cdot \xi_n^+
    - \int_{I_n} A\eta\cdot\xi\,dt.
\]
Sum over $n=0,\dots,N-1$.
\textbf{Left-hand side: use (b) and coercivity.}
By (b), for each $n$,
\[
    \int_{I_n} \xi_t\cdot\xi\,dt
    + \jump{\xi_n}\cdot \xi_n^+
    \;\ge\;
    \frac12|\xi_{n+1}^-|^2 - \frac12|\xi_n^-|^2.
\]
Summing over $n$ telescopes:
\[
    \sum_{n=0}^{N-1}
    \left(
    \int_{I_n} \xi_t\cdot\xi\,dt
    + \jump{\xi_n}\cdot \xi_n^+
    \right)
    \;\ge\;
    \frac12|\xi_N^-|^2 - \frac12|\xi_0^-|^2.
\]
We choose the interpolation $I(u)$ such that
\[
    I(u)(0) = u(0) = u_0,\qquad\text{and }U_0^- = u_0,
\]
hence
\[
    \xi_0^- = U_0^- - I(u)(0) = u_0 - u_0 = 0.
\]
Therefore
\[
    \sum_{n=0}^{N-1}
    \left(
    \int_{I_n} \xi_t\cdot\xi\,dt
    + \jump{\xi_n}\cdot \xi_n^+
    \right)
    \;\ge\;
    \frac12|\xi_N^-|^2.
\]
The other part of the left-hand side is
\[
    \sum_{n=0}^{N-1} \int_{I_n} A\xi\cdot\xi\,dt
    \;\ge\;
    \alpha \sum_{n=0}^{N-1} \int_{I_n} |\xi(t)|^2\,dt
    = \alpha \|\xi\|_{L^2(0,T)}^2.
\]
Altogether,
\[
    \frac12|\xi_N^-|^2 + \alpha \|\xi\|_{L^2(0,T)}^2
    \;\le\; \text{(absolute value of RHS)}.
\]
\textbf{Right-hand side: bound with Cauchy--Schwarz and interpolation estimates.}
For each term, use Cauchy--Schwarz and Young's inequality. For example,
\[
    \left|\int_{I_n} \eta_t\cdot\xi\,dt\right|
    \le \|\eta_t\|_{L^2(I_n)}\,\|\xi\|_{L^2(I_n)}
    \le \frac{\alpha}{4}\|\xi\|_{L^2(I_n)}^2
    + C\|\eta_t\|_{L^2(I_n)}^2,
\]
for some constant $C$ depending on $\alpha$.
Similarly,
\[
    \left|\int_{I_n} A\eta\cdot\xi\,dt\right|
    \le \beta\|\eta\|_{L^2(I_n)}\,\|\xi\|_{L^2(I_n)}
    \le \frac{\alpha}{4}\|\xi\|_{L^2(I_n)}^2
    + C\|\eta\|_{L^2(I_n)}^2.
\]
For the jump term, by standard interpolation construction in time,
$I(u)$ is continuous at the nodes (it interpolates $u(t_n)$), hence
$\jump{\eta_n} = \jump{I(u)-u}_n=0$ and
\[
    \jump{\eta_n}\cdot\xi_n^+ = 0.
\]
Summing over $n$ and absorbing the $\|\xi\|^2$--terms into the left-hand side yields
\[
    \frac12|\xi_N^-|^2 + \frac{\alpha}{2}\|\xi\|_{L^2(0,T)}^2
    \;\le\;
    C\left(
    \|\eta_t\|_{L^2(0,T)}^2 + \|\eta\|_{L^2(0,T)}^2
    \right).
\]
By the interpolation estimate (8) applied to $w=u$ and $w=u_t$ on each $I_n$, we have
(for $u\in H^{k+1}(0,T)$)
\[
    \|\eta\|_{L^2(0,T)}^2 + \|\eta_t\|_{L^2(0,T)}^2
    \;\le\;
    C \tau^{2k+2} |u|_{H^{k+1}(0,T)}^2.
\]
Thus,
\[
    |\xi_N^-|^2 \;\le\; C\tau^{2k+2}|u|_{H^{k+1}(0,T)}^2.
\]
\textbf{Add the interpolation error at $t_N^-$}
At the final time,
\[
    U_N^- - u_N^- = \xi_N^- + \eta_N^-.
\]
Using $|a+b|^2 \le 2|a|^2 + 2|b|^2$,
\[
    |U_N^- - u_N^-|^2
    \le 2|\xi_N^-|^2 + 2|\eta_N^-|^2.
\]
The nodal interpolation property implies
$|\eta_N^-| \le C\tau^{k+1}|u|_{H^{k+1}(0,T)}$, so
\[
    |U_N^- - u_N^-|^2
    \le C\tau^{2k+2}|u|_{H^{k+1}(0,T)}^2.
\]
This proves the desired estimate (9).
\subsection*{(d) Equivalence with backward Euler for $k=0$}
For $k=0$, $V_0^{\mathrm{dG}}$ consists of functions that are \emph{piecewise constant} on each
interval $I_n$.
Consider a single time step: $[0,\tau]$ with $\tau := t_1 - t_0$ and $N=1$.
\begin{itemize}
    \item On $I_0=(0,\tau)$, let $U(t)\equiv U_1$ be constant (so $U_t=0$ a.e. on $(0,\tau)$).
    \item The left value at $t_0$ is $U_0^- = u_0$ (the initial data), and the right value is
          $U_0^+ = U_1$ (upwind value).
    \item The jump at $t_0$ is $\jump{U_0} = U_0^+ - U_0^- = U_1 - u_0$.
\end{itemize}
Take an arbitrary constant test function $V\in V_0^{\mathrm{dG}}$ on $I_0$, so
$V(t)\equiv V_1$.
Apply the dG formulation (6) on $I_0$:
\[
    \int_0^\tau U_t\cdot V\,dt
    + \jump{U_0}\cdot V_0^+
    + \int_0^\tau A U\cdot V\,dt
    = 0.
\]
Since $U_t=0$ a.e.\ on $(0,\tau)$, we get
\[
    (U_1 - u_0)\cdot V_1 + \int_0^\tau A U_1\cdot V_1\,dt = 0.
\]
But $U_1$ and $V_1$ are constant in time, so
\[
    (U_1 - u_0)\cdot V_1 + \tau (A U_1)\cdot V_1 = 0
    \quad\forall V_1\in\mathbb{R}^D.
\]
Thus
\[
    \bigl(U_1 - u_0 + \tau A U_1\bigr)\cdot V_1 = 0
    \quad\forall V_1 \in\mathbb{R}^D
    \quad\Longrightarrow\quad
    U_1 - u_0 + \tau A U_1 = 0.
\]
Rearranging,
\[
    U_1 = u_0 - \tau A U_1.
\]
This is precisely the backward Euler step
\[
    \tilde u_1 = \tilde u_0 - \tau A \tilde u_1, \tag{11}
\]
with identification $\tilde u_0 = u_0$ and $\tilde u_1 = U_1$.
Hence, for $k=0$, the upwind dG method is equivalent to the backward Euler method.
